```python
from __future__ import unicode_literals
import logging
from yt_dlp import YoutubeDL

logger = logging.getLogger(__name__)

def main(input_url):
    yt_opts = {
        "verbose": True,
        "format": "136+ba,298+ba,232+ba,bv+ba",
        "merge_output_format": "mkv",
        "writeautomaticsub": True,
        "subtitlesformat": "srt",
        "noplaylist": True,
        "skip_download": False,
        "outtmpl": "downloads/%(uploader)s/%(title)s.%(ext)s",
        "subtitleslangs": ["en"],
        "writesubtitles": True,
        "writethumbnail": True,
        "postprocessors": [
            { "key": "FFmpegVideoConvertor", "preferedformat": "mkv" },
        ],
    }
    logger.info(f"Starting {input_url}")
    with YoutubeDL(yt_opts) as ydl:
        ydl.download([input_url])
    logger.info(f"Done Downloading {input_url}")
```

**Why it's exemplary:** All configuration in a single dict for maintainability, comprehensive format selection with fallbacks, subtitle and thumbnail downloading, post-processing pipeline.
