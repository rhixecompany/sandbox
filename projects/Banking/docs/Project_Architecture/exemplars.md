Clean singleton pattern for third-party payment API integration with environment-aware configuration.

**Pattern:** Configuration Factory + Client Singleton  
**Category:** Third-Party Integration

```typescript
import { Client } from "dwolla-v2";
import { env } from "@/lib/env";

export type DwollaEnvironment = "production" | "sandbox";

const resolveDwollaEnvironment = (): DwollaEnvironment => {
  const value = env.DWOLLA_ENV ?? "sandbox";
  return value === "production" ? "production" : "sandbox";
};

const resolveDwollaBaseUrl = (environment: DwollaEnvironment): string => {
  return (
    env.DWOLLA_BASE_URL ??
    (environment === "production"
      ? "https://api.dwolla.com"
      : "https://api-sandbox.dwolla.com")
  );
};

export const dwollaConfig = (() => {
  const environment = resolveDwollaEnvironment();
  const baseUrl = resolveDwollaBaseUrl(environment);
  return { baseUrl, environment };
})();

export const getDwollaClient = (): Client => {
  if (!env.DWOLLA_KEY || !env.DWOLLA_SECRET) {
    throw new Error("DWOLLA_KEY and DWOLLA_SECRET are required");
  }
  return new Client({
    environment: dwollaConfig.environment,
    key: env.DWOLLA_KEY,
    secret: env.DWOLLA_SECRET,
  });
};
```

**Why it's exemplary:** IIFE for config singleton, environment resolution with fallback, early validation of required credentials.

```typescript
import { Configuration, PlaidApi, PlaidEnvironments } from "plaid";

export function isMockAccessToken(token: string): boolean {
  if (!token) return false;
  const t = token.toLowerCase();
  return (
    t.startsWith("seed-") || t.startsWith("mock-") ||
    t.startsWith("mock_") || t.startsWith("mock")
  );
}

const plaidEnvironment = env.PLAID_ENV ?? "sandbox";
const configuration = new Configuration({
  baseOptions: {
    headers: {
      "PLAID-CLIENT-ID": env.PLAID_CLIENT_ID,
      "PLAID-SECRET": env.PLAID_SECRET,
    },
  },
  basePath,
});

export const plaidClient = new PlaidApi(configuration);
```

**Why it's exemplary:** Helper function for mock token detection enables development without real Plaid credentials. Clean Configuration setup.
