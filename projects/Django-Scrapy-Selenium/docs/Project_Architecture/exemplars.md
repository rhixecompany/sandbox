import logging
from scrapy.http.request import Request
from scrapy.loader import ItemLoader
from scrapy.spiders import Spider
from scrapy_selenium import SeleniumRequest

class AsuracomicSpider(Spider):
    name = "asuracomic"
    start_urls = list(
        Comic.objects.prefetch_related(
            "comicitems", "comicchapters", "genres", "followers",
        )
        .select_related("author", "type", "artist", "user")
        .all()
        .values_list("url", flat=True)
        .distinct()[0:5],
    )

    def start_requests(self):
        for url in self.start_urls:
            yield Request(url=url, callback=self.comicpage)

    def comicpage(self, response):
        loader = ItemLoader(item=ComicItem(), selector=response)
        loader.add_xpath("title", '//span[contains(@class, "text-xl font-bold")]/text()')
        loader.add_xpath("author", "//div[@class='grid...']/div[2]/h3[2]/text()")
        # ... more structured extraction
        yield loader.load_item()
```

class Bs4Spider(scrapy.Spider):
    name = "bs4_spider"
    start_urls = ["https://asuracomic.net/series/..."]

    def parse(self, response):
        soup = BeautifulSoup(response.text, "html.parser")
        title = soup.find("title").text
        links = [a["href"] for a in soup.find_all("a")]
        images = [img["src"] for img in soup.find_all("img")]
        yield {"title": title, "links": links, "image_urls": images}
```

**Why it's exemplary:** Concise and readable, demonstrates Scrapy + BeautifulSoup interoperability, yields structured items.
