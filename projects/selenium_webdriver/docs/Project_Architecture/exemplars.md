```javascript
async function initializeDriver() {
  let options = new Chrome.Options();
  let directoryName = "chrome-profile";
  let directoryPath = path.join(__dirname, "../", directoryName);
  userDataDir = fs.mkdtempSync(directoryPath, { recursive: true });

  options.addArguments("--headless=new");
  options.addArguments("--disable-gpu");
  options.addArguments("--window-size=1920,1080");
  options.addArguments(`--user-data-dir=${userDataDir}`);
  options.addArguments("--no-sandbox");
  options.addArguments("--disable-dev-shm-usage");
  options.setPageLoadStrategy("eager");
  options.setAcceptInsecureCerts(true);
  options.excludeSwitches("enable-automation");

  driver = await new Builder()
    .forBrowser(Browser.CHROME)
    .setChromeService(service)
    .setChromeOptions(options)
    .build();
}
```

**Why it's exemplary:** Comprehensive Chrome options for reliable headless operation, temporary profile for isolation, eager page load strategy for performance, and insecure cert handling.
